## Overview

The `NotificacionesService` manages in-app notifications within a home. Currently focused on reward redemption notifications ("canjes"), it provides streams for real-time notification updates and methods for marking notifications as seen.

**Location**: `src/app/core/hogar/notificaciones/services/notificaciones.service.ts`

**Injectable**: `{ providedIn: 'root' }`

---

## Creating Notifications

### crearNotificacionCanje()

<ParamField path="crearNotificacionCanje(hogarId, payload)" type="Promise<void>">
  Creates a notification when a user redeems a reward.
</ParamField>

**Parameters**:
<Expandable title="Parameters">
  <ResponseField name="hogarId" type="string" required>
    ID of the home
  </ResponseField>
  <ResponseField name="payload" type="object" required>
    Notification data object
  </ResponseField>
</Expandable>

<Expandable title="payload properties">
  <ResponseField name="actorUid" type="string" required>
    UID of user who redeemed the reward
  </ResponseField>
  <ResponseField name="actorNombre" type="string" required>
    Name of user who redeemed
  </ResponseField>
  <ResponseField name="recompensaId" type="string" required>
    ID of the redeemed reward
  </ResponseField>
  <ResponseField name="recompensaTitulo" type="string" required>
    Title/name of the reward
  </ResponseField>
  <ResponseField name="coste" type="number" required>
    Points cost of the reward
  </ResponseField>
</Expandable>

**Side Effects**:
- Creates notification document in `hogares/{hogarId}/notificaciones`
- Sets `tipo: 'canje'`
- Initializes `vistoPor` map with actor's UID marked as `true`
- Records server timestamp

**Example**:
```typescript
import { NotificacionesService } from '@core/hogar/notificaciones/services/notificaciones.service';

@Component({ ... })
export class RewardsComponent {
  constructor(
    private notificacionesService: NotificacionesService
  ) {}

  async redeemReward(reward: any, currentUser: any) {
    // First, deduct points and mark reward as redeemed
    // ...
    
    // Then create notification
    await this.notificacionesService.crearNotificacionCanje(
      currentUser.hogarId,
      {
        actorUid: currentUser.uid,
        actorNombre: currentUser.nombre,
        recompensaId: reward.id,
        recompensaTitulo: reward.titulo,
        coste: reward.coste
      }
    );
    
    console.log('Reward redeemed and notification created');
  }
}
```

### notificacionesPendientes$()

<ParamField path="notificacionesPendientes$(hogarId, uid)" type="Observable<NotificacionDTO[]>">
  Returns notifications that a specific user hasn't seen yet.
</ParamField>

**Parameters**:
<Expandable title="Parameters">
  <ResponseField name="hogarId" type="string" required>
    ID of the home
  </ResponseField>
  <ResponseField name="uid" type="string" required>
    User ID to check
  </ResponseField>
</Expandable>

**Returns**: `Observable<NotificacionDTO[]>` - Notifications where `vistoPor[uid]` is not `true`

**Example**:
```typescript
import { Component, OnInit } from '@angular/core';
import { NotificacionesService } from '@core/hogar/notificaciones/services/notificaciones.service';
import { AuthService } from '@core/auth/auth.service';

@Component({
  selector: 'app-notification-badge',
  template: `
    <div class="notification-icon" [class.has-pending]="hasPending">
      <i class="bell-icon"></i>
      <span *ngIf="pendingCount > 0" class="badge">
        {{ pendingCount }}
      </span>
    </div>
  `
})
export class NotificationBadgeComponent implements OnInit {
  pendingCount = 0;
  hasPending = false;
  
  constructor(
    private notificacionesService: NotificacionesService,
    private authService: AuthService
  ) {}
  
  ngOnInit() {
    // Assuming you have hogarId available
    const hogarId = 'home-123';
    const uid = this.authService.uidActual!;
    
    this.notificacionesService.notificacionesPendientes$(hogarId, uid)
      .subscribe(pending => {
        this.pendingCount = pending.length;
        this.hasPending = pending.length > 0;
      });
  }
}
```

## Marking as Seen

### marcarComoVista()

<ParamField path="marcarComoVista(hogarId, notiId, uid)" type="Promise<void>">
  Marks a notification as seen by a specific user.
</ParamField>

**Parameters**:
<Expandable title="Parameters">
  <ResponseField name="hogarId" type="string" required>
    ID of the home
  </ResponseField>
  <ResponseField name="notiId" type="string" required>
    ID of the notification
  </ResponseField>
  <ResponseField name="uid" type="string" required>
    User ID marking as seen
  </ResponseField>
</Expandable>

**Side Effects**:
- Updates `vistoPor.{uid}` to `true` in the notification document
- Notification will no longer appear in `notificacionesPendientes$()` for that user

**Example**:
```typescript
async markNotificationAsSeen(notificationId: string) {
  const hogarId = this.currentHogarId;
  const uid = this.authService.uidActual!;
  
  await this.notificacionesService.marcarComoVista(
    hogarId,
    notificationId,
    uid
  );
  
  console.log('Notification marked as seen');
}
```

## Complete Example

```typescript
import { Component, OnInit } from '@angular/core';
import { NotificacionesService } from '@core/hogar/notificaciones/services/notificaciones.service';
import { AuthService } from '@core/auth/auth.service';
import { HogarService } from '@core/hogar/services/hogar.service';

@Component({
  selector: 'app-notifications',
  template: `
    <div class="notifications-panel">
      <h2>
        Notifications
        <span *ngIf="pendingCount > 0" class="badge">
          {{ pendingCount }} new
        </span>
      </h2>
      
      <div *ngFor="let notif of notifications$ | async" 
           class="notification"
           [class.unread]="!notif.vistoPor?.[currentUid]"
           (click)="markAsSeen(notif.id!)">
        
        <div class="notification-content">
          <strong>{{ notif.actorNombre }}</strong> 
          redeemed 
          <strong>{{ notif.recompensaTitulo }}</strong>
          for {{ notif.coste }} points
        </div>
        
        <div class="notification-time">
          {{ notif.createdAt | date:'short' }}
        </div>
      </div>
      
      <div class="stats">
        <p>Total redemptions: {{ totalRedemptions$ | async }}</p>
      </div>
    </div>
  `
})
export class NotificationsComponent implements OnInit {
  notifications$!: Observable<NotificacionDTO[]>;
  pendingCount = 0;
  totalRedemptions$!: Observable<number>;
  currentUid = '';
  hogarId = '';
  
  constructor(
    private notificacionesService: NotificacionesService,
    private authService: AuthService,
    private hogarService: HogarService
  ) {}
  
  ngOnInit() {
    this.currentUid = this.authService.uidActual!;
    
    // Get current home ID
    this.hogarService.getHogar$().subscribe(hogar => {
      if (hogar) {
        this.hogarId = hogar.id!;
        this.setupNotifications();
      }
    });
  }
  
  setupNotifications() {
    // All notifications
    this.notifications$ = this.notificacionesService.notificaciones$(
      this.hogarId
    );
    
    // Pending count
    this.notificacionesService.notificacionesPendientes$(
      this.hogarId,
      this.currentUid
    ).subscribe(pending => {
      this.pendingCount = pending.length;
    });
    
    // Total redemptions
    this.totalRedemptions$ = this.notificacionesService.canjesCount$(
      this.hogarId
    );
  }
  
  async markAsSeen(notificationId: string) {
    await this.notificacionesService.marcarComoVista(
      this.hogarId,
      notificationId,
      this.currentUid
    );
  }
}
```

## Firebase Integration

The service integrates with:
- **Subcollection**: `hogares/{hogarId}/notificaciones`
- **Queries**: Ordered by `createdAt` descending
- **Limits**: Default limits prevent excessive data fetching
- **Real-time**: All observables use `collectionData()` for live updates
